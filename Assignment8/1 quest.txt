#include <stdio.h>
#include <stdlib.h>

struct Node {
    int data;
    struct Node* left;
    struct Node* right;
};

struct Node* createNode(int value) {
    struct Node* newNode = (struct Node*)malloc(sizeof(struct Node));
    newNode->data = value;
    newNode->left = newNode->right = NULL;
    return newNode;
}

struct Node* insert() {
    int value;
    scanf("%d", &value);
    if (value == -1) return NULL;

    struct Node* root = createNode(value);
    root->left = insert();
    root->right = insert();
    return root;
}
